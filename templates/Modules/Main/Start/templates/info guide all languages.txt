                document.getElementById('set_privilege').addEventListener('click', function() {
            // Make an AJAX request to the Flask route
            var xhr = new XMLHttpRequest();
            xhr.open('GET', '/set_privilege', true);
            xhr.onreadystatechange = function () {
                if (xhr.readyState === XMLHttpRequest.DONE) {
                    if (xhr.status === 200) {
                        // On success, reload the page with the register.html content
                        document.open();
                        document.write(xhr.responseText);
                        document.close();
                    } else {
                        // Handle error
                        console.error('Error occurred while loading register page.');
                    }
                }
            };
            xhr.send();
        });
    </script>
